using System.IO;
using System;

public class Definitions
{
    public enum Files
        {
            a = 0,
            b = 1,
            c = 2,
            d = 3,
            e = 4,
            f = 5,
            g = 6,
            h = 7
        }
}

class Program
{
        
    static void Main()
    {
        Console.WriteLine("Hello, World!");
        
        int value = 10;
        
        Files fileFromValue;
        
        if (Enum.IsDefined(typeof(Files), value))
            fileFromValue = (Files)value;
        else
            fileFromValue = Files.a;
        
        Console.WriteLine(fileFromValue);
    }
}
